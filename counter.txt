void decToBinary(int decimal)
{
int binaryNum[32];
int i = 0;
    do  {	
        binaryNum[i] = decimal % 2;
        decimal = decimal / 2;
        i++;
        }
    while (decimal > 0);
    for (int j = i - 1; j >= 0; j--)
        {
cout << binaryNum[j];
        }
}
